source 就是 source as T，把 source 断言成 T 类型

function createArray(length: number, value: T): Array{
let result: T[] = [];
for (let i = 0; i < length; i++) {
result[i] = value;
}
return result;
}

createArray(3, ‘x’); // [‘x’, ‘x’, ‘x’]
